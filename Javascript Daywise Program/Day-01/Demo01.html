Javascript Code to calculate:
Sum, Substraction, Multiplication, Remainder and Quotient?

                
<script type="text/javascript">  

    function calculate() {
        var num1 = parseFloat(document.getElementById('num1').value);
        var num2 = parseFloat(document.getElementById('num2').value);
                            
        var sum = num1 + num2;
        var sub = num1 - num2;
        var mul = num1 * num2;
        var rem = num1 % num2;
        var quotient = num1 / num2;
                            
        document.getElementById('result-sum').innerHTML = 'Sum: ' + sum;
        document.getElementById('result-sub').innerHTML = 'Subtraction: ' + sub;
        document.getElementById('result-mul').innerHTML = 'Multiplication: ' + mul;
        document.getElementById('result-rem').innerHTML = 'Remainder: ' + rem;
        document.getElementById('result-quotient').innerHTML = 'Quotient: ' + quotient; 
    }

</script>
