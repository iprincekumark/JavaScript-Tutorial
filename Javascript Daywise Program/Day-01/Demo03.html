Javascript Code to Swap two numbers?

                
<script type="text/javascript">  

    function beforeSwap() {
        var num1 = parseInt(document.getElementById("num1").value);
        var num2 = parseInt(document.getElementById("num2").value);

        document.getElementById('breforeSwap').innerHTML = "Before Swap 
" + "num1 = " + num1 + "
" + "num2 = " + num2 ;
    }

    function afterSwap() {
        var num1 = parseInt(document.getElementById("num1").value);
        var num2 = parseInt(document.getElementById("num2").value);

        var temp = num1;
        num1 = num2;
        num2 = temp;

        document.getElementById('afterSwap').innerHTML = "After Swap 
" + "num1 = " + num1 + "
" + "num2 = " + num2;
    }

</script>
