Javascript Code to calculate with seperate button for each operation:
Sum, Substraction, Multiplication, Remainder and Quotient?

                
<script type="text/javascript">  

    function calculateSum() {
            var num1 = parseFloat(document.getElementById('num1').value);
            var num2 = parseFloat(document.getElementById('num2').value);
            var sum = num1 + num2;
            document.getElementById('result').innerHTML = 'Sum: ' + sum;
        }

        function calculateSubtract() {
            var num1 = parseFloat(document.getElementById('num1').value);
            var num2 = parseFloat(document.getElementById('num2').value);
            var sub = num1 - num2;
            document.getElementById('result').innerHTML = 'Subtraction: ' + sub;
        }

        function calculateMultiply() {
            var num1 = parseFloat(document.getElementById('num1').value);
            var num2 = parseFloat(document.getElementById('num2').value);
            var mul = num1 * num2;
            document.getElementById('result').innerHTML = 'Multiplication: ' + mul;
        }

        function calculateModulus() {
            var num1 = parseFloat(document.getElementById('num1').value);
            var num2 = parseFloat(document.getElementById('num2').value);
            var rem = num1 % num2;
            document.getElementById('result').innerHTML = 'Remainder: ' + rem;
        }

        function calculateQuotient() {
            var num1 = parseFloat(document.getElementById('num1').value);
            var num2 = parseFloat(document.getElementById('num2').value);
            var quotient = num1 / num2;
            document.getElementById('result').innerHTML = 'Quotient: ' + quotient;
        }

</script>
